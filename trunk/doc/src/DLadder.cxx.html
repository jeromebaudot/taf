void DLadder::UpdateAlignment()
{
  
  // Update the positions (tilts and translations) of all planes within the ladder.
  //
  // LC
  // Modified: JB 2014/02/17 Use ConvoluteAlignment for Custom case
  
  if (fDebugLadder) {
    printf("  DLadder %2d,  updating alignment for %d planes.\n",ladderID,GetNumberOfPlanes());
  }

  ladderPosition(0) = (Float_t)fPrecAlign->GetTranslation()[0];
  ladderPosition(1) = (Float_t)fPrecAlign->GetTranslation()[1];
  ladderPosition(2) = (Float_t)fPrecAlign->GetTranslation()[2];
  
  ladderRotation(0) = (Float_t)fPrecAlign->GetRotations()[2];
  ladderRotation(1) = (Float_t)fPrecAlign->GetRotations()[1];
  ladderRotation(2) = (Float_t)fPrecAlign->GetRotations()[0];
  
  (*fTilt)(0) = (Float_t)fPrecAlign->GetRotations()[2];
  (*fTilt)(1) = (Float_t)fPrecAlign->GetRotations()[1];
  (*fTilt)(2) = (Float_t)fPrecAlign->GetRotations()[0];
  
  (*fPosition)(0) = (Float_t)fPrecAlign->GetTranslation()[0];
  (*fPosition)(1) = (Float_t)fPrecAlign->GetTranslation()[1];
  (*fPosition)(2) = (Float_t)fPrecAlign->GetTranslation()[2];
  
  //ladderPosition.Print();
  //ladderRotation.Print();
  //std::map<Int_t, DR3>::iterator iter = planePosition.begin();
  
  for(Int_t i=1; i<=GetNumberOfPlanes() ; ++i) { // loop on planes
    
    // DR3 position = planePosition[i];
    // planeList[i]->TrackerUpdate();
    
    DR3 newRotations(ladderRotation(2)-ladderInitialRotations(2)+planeRotation[i](2), ladderRotation(1)-ladderInitialRotations(1)+planeRotation[i](1), ladderRotation(0)-ladderInitialRotations(0)+planeRotation[i](0) );
    
    if(ladderType=="Salat" || ladderType=="salat" || ladderType=="Plume" || ladderType=="plume") planeList[i]->GetPrecAlignment()->SetRotations(newRotations); 
    
    DR3 ladderPositionCorrection(0,0,0);
    if(ladderType=="Salat"){
      
      if(i==2 || i==4) ladderPositionCorrection(0) = + ladderPosition(0);
      else if (i==1 || i==3) ladderPositionCorrection(0) = - ladderPosition(0);
      
      if(i==1 || i==2) ladderPositionCorrection(1) = - ladderPosition(1);
      else if (i==3 || i==4) ladderPositionCorrection(1) = + ladderPosition(1);
      ladderPositionCorrection(2) = ladderPosition(2);
    }
    else ladderPositionCorrection = ladderPosition;
    
    if(ladderType!="Custom" && ladderType!="custom") {
      planeList[i]->GetPrecAlignment()->SetTranslation( ladderPositionCorrection + planePosition[i] );
      //planePosition[i].Print();
    }
    
    else { // for Custom case, JB 2014/02/17
      
      // Set the alignment relative to the ladder
      planeList[i]->GetPrecAlignment()->CopyAlignment( &(planeRelativePrecAlign[i]) );
      if( fDebugLadder ) {
        cout << " DLadder::UpdateAlignement: Plane alignement before convolution" << endl;
        planeList[i]->GetPrecAlignment()->PrintAll();                                        
      }
      // Convolute the relative alignment with the ladder alignment
      planeList[i]->GetPrecAlignment()->ConvoluteAlignment( fPrecAlign );
      
    }

    planeList[i]->UpdatePositions();
    planeList[i]->GetPrecAlignment()->PrintAll();                                        

  }  // end loop on planes
  
}

void DLadder::GetRelativePosition( Int_t aPlaneNumber, DR3* aTilt, DR3* aShift)
{
  
  // Set the provided tilts and translations pointers to the relative positions 
  // of the given planes with respect to the ladder.
  // Note aPlaneNumber is the relative number within the ladder (from 1)!
  //
  // JB 2014/02/10
  
  if( 0<aPlaneNumber && aPlaneNumber<GetNumberOfPlanes() ) {
    std::cout << "WARNING there are only " << GetNumberOfPlanes() << " planes in ladder " << ladderID << "(" << GetPlaneType().Data() << ")!" << std::endl; 
    return;
  }
  
  if(ladderType!="Custom") {
    ladderPosition(0) = (Float_t)fPrecAlign->GetTranslation()[0];
    ladderPosition(1) = (Float_t)fPrecAlign->GetTranslation()[1];
    ladderPosition(2) = (Float_t)fPrecAlign->GetTranslation()[2];
    
    ladderRotation(0) = (Float_t)fPrecAlign->GetRotations()[2];
    ladderRotation(1) = (Float_t)fPrecAlign->GetRotations()[1];
    ladderRotation(2) = (Float_t)fPrecAlign->GetRotations()[0];
    
    Double_t *planeRotation = planeList[aPlaneNumber]->GetPrecAlignment()->GetRotations();
    Double_t *planeTranslation = planeList[aPlaneNumber]->GetPrecAlignment()->GetTranslation();
    
    aTilt->SetValue( ladderRotation(2)-planeRotation[2]
                    ,ladderRotation(1)-planeRotation[1]
                    ,ladderRotation(0)-planeRotation[0] );
    
    aShift->SetValue( ladderPosition(0)-planeTranslation[0]
                     ,ladderPosition(1)-planeTranslation[1]
                     ,ladderPosition(2)-planeTranslation[2] );  
  }
  
  else { // for Custom case
    
    aTilt->SetValue( (Float_t*)(planeRelativePrecAlign[aPlaneNumber].GetRotations()) );
    aShift->SetValue( (Float_t*)(planeRelativePrecAlign[aPlaneNumber].GetTranslation()) );
    
  }
  
  
  printf( " Relative position of plane %d (absolute #=%d) wrt ladder %d:\n", aPlaneNumber, planeList[aPlaneNumber]->GetPlaneNumber(), ladderID);
  printf( "   Tilt: /z=%.2f, /y=%.2f, /x=%.2f\n", (*aTilt)(0), (*aTilt)(1), (*aTilt)(2));
  printf( "   Shift: z=%.2f, y=%.2f, x=%.2f\n", (*aShift)(0), (*aShift)(1), (*aShift)(2));

}

void DLadder::MakeTrack(DR3& positionLeft, DR3& positionRight, DTrack* myTrack)
{
   
  Float_t x0 = positionLeft(0);
  Float_t y0 = positionLeft(1);
  Float_t z0 = positionLeft(2);

  Float_t x1 = positionRight(0);
  Float_t y1 = positionRight(1);
  Float_t z1 = positionRight(2);

 //-----------------------------//
 //       x = az + b            //
 //       y = cz + d            //
 //-----------------------------//
  
  DR3 tLineFitOrigin(0.,0.,0.);  // Origin in the tracker system
  DR3 tLineFitSlope(0.,0.,0.);   // Slope in the tracker system
 
  tLineFitOrigin.SetValue(0.,0.,0.);
  tLineFitSlope.SetValue(0.,0.,1.);

  if( z1-z0 != 0. ) {
    tLineFitOrigin(0) = (z1*x0-z0*x1)/(z1-z0);
    tLineFitOrigin(1) = (z1*y0-z0*y1)/(z1-z0);
    tLineFitSlope(0)  = (x1-x0)/(z1-z0);
    tLineFitSlope(1)  = (y1-y0)/(z1-z0);
  }
  else {
    tLineFitOrigin(0) = x1;
    tLineFitOrigin(1) = y1; 
    tLineFitSlope(0)  = 0.;
    tLineFitSlope(1)  = 0.;      
  }

  //std::cout<<"tLineFitSlope = "<<tLineFitSlope(0)<<"   "<<tLineFitSlope(1)<<std::endl;

  DLine* aLine = new DLine();
  aLine->SetValue(tLineFitOrigin, DR3(0.,0.,0.) , tLineFitSlope, 0.);

  myTrack->SetLinearFit(aLine);

  //std::cout<< myTrack->GetLinearFit().GetSlopeZ()(0)<<"   "<<myTrack->GetLinearFit().GetSlopeZ()(1)<<std::endl;
}

  fhDW->Fill(Data[12]); // 12 -> DW CoG

  fhDU_U->Fill(Data[0],Data[7]);  // DU vs U 
  fhDV_V->Fill(Data[1],Data[8]);  // DV vs V

  fhDU_V->Fill(Data[0],Data[8]);  // DU vs V
  fhDV_U->Fill(Data[1],Data[7]);  // DU vs U

  Float_t angle = 180./TMath::Pi()*TMath::ATan( TMath::Sqrt(dxdydz(0)*dxdydz(0)+dxdydz(1)*dxdydz(1)) / dxdydz(2) );
  fhDiffSlopeX->Fill( (angle-angleMC)/angleMC );

}

double DLadder::RotationCoordX(double x, double y, double z, double angleX, double angleY, double angleZ)
{
  angleX = - angleX;
  angleY = - angleY; 
  angleZ = - angleZ;

  double Xprime = x*cos(angleY)*cos(angleZ) + y*( sin(angleX)*sin(angleY)*cos(angleZ)-cos(angleX)*sin(angleZ) ) + z*( cos(angleX)*sin(angleY)*cos(angleZ) + sin(angleZ)*sin(angleX) );  
  //G4double Xprime2 = x*cos(angleX)*cos(angleY) + y*( sin(angleZ)*sin(angleY)*cos(angleX)-cos(angleZ)*sin(angleX) ) + z*( cos(angleX)*sin(angleY)*cos(angleZ) + sin(angleX)*sin(angleZ) );
  return Xprime;
}


//______________________________________________________________________________
//  
double DLadder::RotationCoordY(double x, double y, double z, double angleX, double angleY, double angleZ)
{
  angleX = - angleX;
  angleZ = - angleZ;
  angleY = - angleY;
  
  double Yprime = x*cos(angleY)*sin(angleZ) + y*( cos(angleX)*cos(angleZ)+sin(angleX)*sin(angleY)*sin(angleZ) ) + z*( cos(angleX)*sin(angleY)*sin(angleZ)-sin(angleX)*cos(angleZ) );
  //G4double Yprime2 = x*cos(angleY)*sin(angleX) + y*( cos(angleX)*cos(angleZ)+sin(angleX)*sin(angleY)*sin(angleZ) ) + z*( cos(angleZ)*sin(angleX)*sin(angleY)-sin(angleZ)*cos(angleX) );
  return Yprime;
}


//______________________________________________________________________________
//  
double DLadder::RotationCoordZ(double x, double y, double z, double angleX, double angleY, double angleZ)
{
  angleX = - angleX;
  angleY = - angleY;
  angleZ = - angleZ;
  
  double Zprime = -x*sin(angleY) + y*sin(angleY)*cos(angleX) + z*cos(angleX)*cos(angleY);
  //G4double Zprime2 = -x*sin(angleY) + y*sin(angleZ)*cos(angleY) + z*cos(angleY)*cos(angleZ);
  return Zprime;
}
